 55    def _process_output(
 56        self,
 57        stderr_line: str,
 58        stderr: List[str],
 59        duration_override: Union[float, None],
 60    ) -> Union[float, None]:
 61        """
 62        Process the output of the ffmpeg command.
 63
 64        Args:
 65            stderr_line (str): The line of stderr output.
 66            stderr (List[str]): The list of stderr output.
 67            duration_override (Union[float, None]): The duration of the video in seconds.
 68
 69        Returns:
 70            Union[float, None]: The progress in percent.
 71        """
 72
 73        if self.stderr_callback:
 74            self.stderr_callback(stderr_line)
 75
 76        stderr.append(stderr_line.strip())
 77        self.stderr = "\n".join(stderr)
 78
 79        progress: Union[float, None] = None
 80        # assign the total duration if it was found. this can happen multiple times for multiple inputs,
 81        # in which case we have to determine the overall duration by taking the min/max (dependent on -shortest being present)
 82        if (
 83            current_dur_match := self.DUR_REGEX.search(stderr_line)
 84        ) and duration_override is None:
 85            input_options = self.inputs_with_options[self.current_input_idx]
 86            current_dur_ms: int = to_ms(**current_dur_match.groupdict())
 87            # if the previous line had "image2", it's a single image and we assume a really short intrinsic duration (4ms),
 88            # but if it's a loop, we assume infinity
 89            if "image2" in stderr[-2] and "-loop 1" in " ".join(input_options):
 90                current_dur_ms = 2**64
 91            if "-shortest" in self.cmd:
 92                self.total_dur = (
 93                    min(self.total_dur, current_dur_ms)
 94                    if self.total_dur is not None
 95                    else current_dur_ms
 96                )
 97            else:
 98                self.total_dur = (
 99                    max(self.total_dur, current_dur_ms)
100                    if self.total_dur is not None
101                    else current_dur_ms
102                )
103            self.current_input_idx += 1
104
105        if (
106            progress_time := self.TIME_REGEX.search(stderr_line)
107        ) and self.total_dur is not None:
108            elapsed_time = to_ms(**progress_time.groupdict())
109            progress = min(max(round(elapsed_time / self.total_dur * 100, 2), 0), 100)
110
111        return progress

114    def _probe_duration(cmd: List[str]) -> Optional[int]:
115        """
116        Get the duration via ffprobe from input media file
117        in case ffmpeg was run with loglevel=error.
118
119        Args:
120            cmd (List[str]): A list of command line elements, e.g. ["ffmpeg", "-i", ...]
121
122        Returns:
123            Optional[int]: The duration in milliseconds.
124        """
125        file_names = []
126        for i, arg in enumerate(cmd):
127            if arg == "-i":
128                file_name = cmd[i + 1]
129
130                # filter for filenames that we can probe, i.e. regular files
131                if os.path.isfile(file_name):
132                    file_names.append(file_name)
133
134        if len(file_names) == 0:
135            return None
136
137        durations = []
138
139        for file_name in file_names:
140            try:
141                output = subprocess.check_output(
142                    [
143                        "ffprobe",
144                        "-loglevel",
145                        "error",
146                        "-hide_banner",
147                        "-show_entries",
148                        "format=duration",
149                        "-of",
150                        "default=noprint_wrappers=1:nokey=1",
151                        file_name,
152                    ],
153                    universal_newlines=True,
154                )
155                durations.append(int(float(output.strip()) * 1000))
156            except Exception:
157                # TODO: add logging
158                return None
159
160        return max(durations) if "-shortest" not in cmd else min(durations)

173    @staticmethod
174    def _get_inputs_with_options(cmd: List[str]) -> List[List[str]]:
175        """
176        Collect all inputs with their options.
177        For example, input is:
178
179            ffmpeg -i input1.mp4 -i input2.mp4 -i input3.mp4 -filter_complex ...
180
181        Output is:
182
183            [
184                ["-i", "input1.mp4"],
185                ["-i", "input2.mp4"],
186                ["-i", "input3.mp4"],
187            ]
188
189        Another example:
190
191            ffmpeg -f lavfi -i color=c=black:s=1920x1080 -loop 1 -i image.png -filter_complex ...
192
193        Output is:
194
195            [
196                ["-f", "lavfi", "-i", "color=c=black:s=1920x1080"],
197                ["-loop", "1", "-i", "image.png"],
198            ]
199        """
200        inputs = []
201        prev_index = 0
202        for i, arg in enumerate(cmd):
203            if arg == "-i":
204                inputs.append(cmd[prev_index : i + 2])
205                prev_index = i + 2
206
207        return inputs

209    def run_command_with_progress(
210        self, popen_kwargs=None, duration_override: Union[float, None] = None
211    ) -> Iterator[float]:
212        """
213        Run an ffmpeg command, trying to capture the process output and calculate
214        the duration / progress.
215        Yields the progress in percent.
216
217        Args:
218            popen_kwargs (dict, optional): A dict to specify extra arguments to the popen call, e.g. { creationflags: CREATE_NO_WINDOW }
219            duration_override (float, optional): The duration in seconds. If not specified, it will be calculated from the ffmpeg output.
220
221        Raises:
222            RuntimeError: If the command fails, an exception is raised.
223
224        Yields:
225            Iterator[float]: A generator that yields the progress in percent.
226        """
227        if self.dry_run:
228            yield from [0, 100]
229            return
230
231        if duration_override:
232            self.total_dur = int(duration_override * 1000)
233
234        base_popen_kwargs = self.base_popen_kwargs.copy()
235        if popen_kwargs is not None:
236            base_popen_kwargs.update(popen_kwargs)
237
238        self.process = subprocess.Popen(self.cmd_with_progress, **base_popen_kwargs)  # type: ignore
239
240        yield 0
241
242        stderr: List[str] = []
243        while True:
244            if self.process.stdout is None:
245                continue
246
247            stderr_line: str = (
248                self.process.stdout.readline().decode("utf-8", errors="replace").strip()
249            )
250
251            if stderr_line == "" and self.process.poll() is not None:
252                break
253
254            progress = self._process_output(stderr_line, stderr, duration_override)
255            if progress is not None:
256                yield progress
257
258        if self.process.returncode != 0:
259            raise RuntimeError(f"Error running command {self.cmd}: {self.stderr}")
260
261        yield 100
262        self.process = None

383    def set_stderr_callback(self, callback: Callable[[str], None]) -> None:
384        """
385        Set a callback function to be called on stderr output.
386        The callback function must accept a single string argument.
387        Note that this is called on every line of stderr output, so it can be called a lot.
388        Also note that stdout/stderr are joined into one stream, so you might get stdout output in the callback.
389
390        Args:

392        """

394            raise ValueError(

396            )

398        self.stderr_callback = callback
